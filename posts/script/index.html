#!/bin/bash

# Variables ruta_pagina="/home/usuario/pineweb/"
public="public/*" respositorio_public="/home/usuario/github/public/"

# Cambiar al directorio de la página cd "$ruta_pagina" || { echo "Error: No se pudo cambiar al directorio $ruta_pagina"; exit 1; }

#Creamos la pagina estatica con el comando hugo hugo > /dev/null 2>&1

#Copiamos el contenido de la carpeta public y la pegamos en el repositorio

rsync -av --delete $public $respositorio_public > /dev/null 2>&1

#Realizamos un commit de la actualizacion de contenido y lo subimos

cd $respositorio_public git add . git commit -am "Se ha actualizado el contenido " > /dev/null 2>&1 git push > /dev/null 2>&1
